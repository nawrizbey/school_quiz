
3.5. DETERMINISTIC SHUFFLE ALGORITMI

MUAMMO:
- Har bir o'quvchi uchun javoblar aralashtirilishi kerak
- Lekin bir xil o'quvchi qayta kirganda bir xil tartib bo'lishi kerak
- To'g'ri javob indeksi saqlanishi kerak

SEEDED RANDOM ALGORITHM:
```typescript
// Seed generator (student_name + quiz_id dan)
function generateSeed(studentName: string, quizId: number): number {
  let hash = 0;
  const str = `${studentName}_${quizId}`;

  for (let i = 0; i < str.length; i++) {
    const char = str.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash; // 32-bit integer
  }

  return Math.abs(hash);
}

// Seeded random number generator
function seededRandom(seed: number) {
  let value = seed;
  return function() {
    value = (value * 9301 + 49297) % 233280;
    return value / 233280;
  };
}
```

FISHER-YATES SHUFFLE:
```typescript
function shuffleOptions(
  question: Question,
  studentName: string,
  quizId: number
): { options: string[], correctIndex: number } {

  const seed = generateSeed(studentName, quizId) + question.id;
  const random = seededRandom(seed);

  // Javoblar massivi
  const options = [
    question.option_a,
    question.option_b,
    question.option_c,
    question.option_d
  ];

  // Asl to'g'ri javob indeksi
  const originalCorrectIndex = question.correct_option;

  // Fisher-Yates shuffle
  const shuffled = [...options];
  const indexMap = [0, 1, 2, 3];

  for (let i = shuffled.length - 1; i > 0; i--) {
    const j = Math.floor(random() * (i + 1));

    // Swap elements
    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
    [indexMap[i], indexMap[j]] = [indexMap[j], indexMap[i]];
  }

  // Yangi to'g'ri javob indeksi
  const newCorrectIndex = indexMap.indexOf(originalCorrectIndex);

  return { options: shuffled, correctIndex: newCorrectIndex };
}
```

```typescript
    question,
    studentName,
    quizId
  );

  return {
    ...question,
    shuffledOptions: options,
    shuffledCorrectIndex: correctIndex
  };
});
```

```typescript
async function uploadImage(file: File): Promise<string> {
  // 1. File validation
  const maxSize = 5 * 1024 * 1024; // 5MB
  if (file.size > maxSize) {
    throw new Error('File size must be less than 5MB');
  }

  const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
  if (!allowedTypes.includes(file.type)) {
    throw new Error('Only JPEG, PNG, GIF, and WebP images are allowed');
  }

  // 2. Generate unique filename
  const fileExt = file.name.split('.').pop();
  const fileName = `${Date.now()}_${Math.random().toString(36).substring(7)}.${fileExt}`;

  // 3. Upload to Supabase Storage
  const { data, error } = await supabase.storage
    .from('quiz-images')
    .upload(fileName, file, {
      cacheControl: '3600',
      upsert: false
    });

  if (error) throw error;

  // 4. Get public URL
  const { data: { publicUrl } } = supabase.storage
    .from('quiz-images')
    .getPublicUrl(fileName);

  return publicUrl;
}
```

```typescript
// Dashboard da real-time updates
useEffect(() => {
  const channel = supabase
    .channel('quiz-changes')
    .on(
      'postgres_changes',
      {
        event: '*', // INSERT, UPDATE, DELETE
        schema: 'public',
        table: 'quizzes'
      },
      (payload) => {
        console.log('Change received!', payload);

        if (payload.eventType === 'INSERT') {
          // Yangi test qo'shildi
          setQuizzes(prev => [payload.new, ...prev]);
        } else if (payload.eventType === 'UPDATE') {
          // Test yangilandi
          setQuizzes(prev => prev.map(q =>
            q.id === payload.new.id ? payload.new : q
          ));
        } else if (payload.eventType === 'DELETE') {
          // Test o'chirildi
          setQuizzes(prev => prev.filter(q => q.id !== payload.old.id));
        }
      }
    )
    .subscribe();

  return () => {
    supabase.removeChannel(channel);
  };
}, []);
```
